\begin{exercise}


  Consider the matrices:
  \begin{equation*}
    A = \startmat{cc}
      1 & 2 \\
      3 & 2 \\
      1 & -1
    \stopmat,
    \quad
    B = \startmat{ccc}
      2 & -5 & 2 \\
      -3 & 2 & 1
    \stopmat
  \end{equation*}
  Find the following if possible.
  \begin{enumerate}
  \item $-3A{^T}$.
  \item $3B - A^T$.
  \end{enumerate}

  $-3A^T$ \wordChoice{
    \choice[correct]{is}
    \choice{is not}
  } possible because

  \begin{feedback}
    the transpose of a matrix is another matrix, and multiplying a matrix by a scalar is possible.
  \end{feedback}


$\startmat{ccc}
        \answer{-3} & -9 & \answer{-3} \\
        -6 & \answer{-6} & 3
      \stopmat$.


$3B - A^T$ \wordChoice{
  \choice[correct]{is}
  \choice{is not}
} possible because

  \begin{problem}
\begin{selectAll}
    \choice[correct]{the transpose of a matrix is another matrix, and multiplying a matrix by a scalar is possible.}
    \choice[correct]{the dimensions of $A^T$ and $B$ are the same.}
    \choice{the dimensions of $A^T$ and $B$ are different.}
    \choice{the dimensions of $A^T$ and $3B$ are different.}
    \choice{the dimensions of $A^T$ and $3B$ are the different.}
\end{selectAll}

$\startmat{ccc}
        \answer{5} & -18 & \answer{5} \\
        -11 & \answer{4} & 4
      \stopmat$.
      \end{problem}
