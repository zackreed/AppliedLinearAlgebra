\begin{exercise}

Which are unit vectors? (Remember to answer in decimals up to two decimal places, unless you are entering an integer.)
\begin{equation*}
  \vect{u}=\frac{1}{2}\startmat{c}1\\1\\1\stopmat, \quad
  \vect{v}=\frac{1}{3}\startmat{c}1\\1\\1\stopmat, \quad
  \vect{w}=\frac{1}{2}\startmat{c}1\\1\\1\\1\stopmat.
\end{equation*}

$\vect{u}$ \wordChoice{
  \choice{is}
  \choice[correct]{isn't}
  } a unit vector with norm $\approx\answer[tolerance=.1]{.87}$. $\vect{v}$ \wordChoice{
    \choice{is}
    \choice[correct]{isn't}
    } a unit vector with norm $\approx\answer[tolerance=.1]{.58}$. $\vect{w}$ \wordChoice{
      \choice[correct]{is}
      \choice{isn't}
      } a unit vector with norm $\approx\answer{1}$.

\end{exercise}
