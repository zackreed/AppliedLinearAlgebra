\begin{exercise}

Determine whether the following systems of equations have solutions and why and how many. If they do have solutions, find the solution.

(NOTE: There are follow up questions that appear once these multiple choice questions are correctly answered.)

\begin{enumerate}

\item  \begin{equation*}
    \begin{array}{c}
      3x+y=3 \\
      x+2y=1.
    \end{array}
  \end{equation*} has 

  \begin{multipleChoice}
    \choice{no solutions}
    \choice[correct]{one solution}
    \choice{infinitely many solutions}
  \end{multipleChoice}

  \begin{problem}
  %\begin{sol}
    $(x,y) = (\answer{1},\answer{0})$.
  %\end{sol}
    \end{problem}

\item $
  \begin{array}{c}
    x+3y=1 \\
    4x-y=3
  \end{array}
  $ has

    \begin{multipleChoice}
        \choice{no solutions}
        \choice[correct]{one solution}
        \choice{infinitely many solutions}
    \end{multipleChoice}
  
  \begin{problem}
  
    $(x,y) = (\answer{10/13},\answer{1/13})$

  \end{problem}

\item $
\begin{array}{c}
    x+2y=1 \\
    2x-y=1 \\
    4x+3y=3
  \end{array}
  $, has
  
  \begin{multipleChoice}
    \choice{no solutions}
    \choice[correct]{one solution}
    \choice{infinitely many solutions}
    \end{multipleChoice}
  
  \begin{problem}
    $(x,y) = (\answer{3/5},\answer{1/5})$
    \end{problem}

\item $
    \begin{array}{c@{~}c@{~}c@{~}c@{~}c@{~}c@{~}c}
      x &+& y &-& 3z &=& 2 \\
      2x &+& y &+& z &=& 1 \\
      3x &+& 2y &-& 2z &=& 0
    \end{array}
    $ has

    \begin{multipleChoice}
        \choice[correct]{no solutions}
        \choice{one solution}
        \choice{infinitely many solutions}
    \end{multipleChoice}

    \begin{problem}
        because

        \begin{multipleChoice}
        
            \choice[correct]{the system reduces one equation to $0=1$}
            \choice{two planes are parallel and not the same}
            \choice[correct]{the rows are linearly dependent}
            \choice{two planes are parallel and are the same}
            \choice{all three planes are the same and parallel}
            \choice{no two planes are parallel}

        \end{multipleChoice}
    \end{problem}


\end{enumerate}

\end{exercise}
