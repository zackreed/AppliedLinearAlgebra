  \begin{problem}
    Find the area of the parallelogram determined by the vectors
    $\startmat{r}
      1 \\
      2 \\
      3
    \stopmat$, $\startmat{r}
      3 \\
      -2 \\
      1
    \stopmat$.

      $\startmat{r}
        1 \\
        2 \\
        3
      \stopmat \times
      \startmat{r}
        3 \\
        -2 \\
        1
      \stopmat = \startmat{r}
        \answer{8} \\
        \answer{8} \\
        \answer{-8}
      \stopmat$. 
      
      The area of the parallelogram is $\answer[tolerance=.5]{8\sqrt{3}}$.
    
  \end{problem}
