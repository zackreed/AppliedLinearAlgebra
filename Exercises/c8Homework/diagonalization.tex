\begin{exercise}
  For each of the following symmetric matrices, find the eigenvalues,
  an orthonormal basis for each eigenspace, and then orthogonally
  diagonalize the matrix.
  \begin{equation*}
    (a)\quad
    A = \startmat{rr}
      -1 & 2 \\
      2  & 2 \\
    \stopmat,\quad
    (b)\quad
    A = \startmat{rr}
      3 & 1 \\
      1 & 3 \\
    \stopmat,\quad
    (c)\quad
    A = \startmat{rrr}
      1 & 1 & 0 \\
      1 & 2 & 1 \\
      0 & 1 & 1 \\
    \stopmat,\quad
    (d)\quad
    A = \startmat{rrr}
      2 & 1 & 2 \\
      1 & 2 & 2 \\
      2 & 2 & 5 \\
    \stopmat.
  \end{equation*}
  \begin{solution}
    \begin{enumerate}
    \item $A=PDP^{-1}$ where $D=\startmat{rr}
        -2 & 0 \\
        0  & 3 \\
      \stopmat$,
      $P=\frac{1}{\sqrt{5}}\startmat{rr}
        2  & 1 \\
        -1 & 2 \\
      \stopmat$.
    \item $A=PDP^{-1}$ where $D=\startmat{rr}
        2 & 0 \\
        0 & 4 \\
      \stopmat$,
      $P=\frac{1}{\sqrt{2}}\startmat{rr}
        1  & 1 \\
        -1 & 1 \\
      \stopmat$.
    \item $A=PDP^{-1}$ where
      $D=\startmat{rrr}
        0 & 0 & 0 \\
        0 & 1 & 0 \\
        0 & 0 & 3 \\
      \stopmat$,
      $\def\arraystretch{1.3}
      P=\startmat{rrr}
        \frac{1}{\sqrt{3}} & \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{6}} \\
        -\frac{1}{\sqrt{3}} & 0 & \frac{2}{\sqrt{6}} \\
        \frac{1}{\sqrt{3}} & -\frac{1}{\sqrt{2}} & \frac{1}{\sqrt{6}} \\
      \stopmat$.
    \item $A=PDP^{-1}$ where
      $D=\startmat{rrr}
        1 & 0 & 0 \\
        0 & 1 & 0 \\
        0 & 0 & 7 \\
      \stopmat$,
      $\def\arraystretch{1.3}
      P=\startmat{rrr}
        \frac{1}{\sqrt{2}} & \frac{1}{\sqrt{3}} & \frac{1}{\sqrt{6}} \\
        -\frac{1}{\sqrt{2}} & \frac{1}{\sqrt{3}} & \frac{1}{\sqrt{6}} \\
        0 & -\frac{1}{\sqrt{3}} & \frac{2}{\sqrt{6}} \\
      \stopmat$.
    \end{enumerate}
  \end{solution}
\end{exercise}
