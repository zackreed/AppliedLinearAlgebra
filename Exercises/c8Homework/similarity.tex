  \begin{problem}
  By computing the trace, determinant, and rank, show that $A$ and $B$ are \textit{not} similar in each case.
   
  \begin{problem}\label{prob:notsimilar_a}
  $A = \begin{bmatrix}
      1 & 2 \\
      2 & 1
  \end{bmatrix}$, $B =
  \begin{bmatrix}
      1 & 1\\
      -1 & 1
  \end{bmatrix}$
  \end{problem}
   
  \begin{problem}\label{prob:notsimilar_b}
  $A =
  \begin{bmatrix}
  3 & 1 \\
  2 & -1
  \end{bmatrix}$, $B =
  \begin{bmatrix}
  1 & 1 \\
  2 & 1
  \end{bmatrix}$
  \end{problem}
   
  \begin{problem}\label{prob:notsimilar_c}
  $A =
  \begin{bmatrix}
  3 & 1 \\
  -1 & 2
  \end{bmatrix}$, $B =
  \begin{bmatrix}
  2 & -1 \\
  3 & 2
  \end{bmatrix}$
  \end{problem}
   
  \begin{problem}\label{prob:notsimilar_d}
  $A =
  \begin{bmatrix}
  2 & 1 & 1 \\
  1 & 0 & 1 \\
  1 & 1 & 0
  \end{bmatrix}$, $B =
  \begin{bmatrix}
   1 & -2 & 1 \\
  -2 & 4 & -2 \\
  -3 & 6 & -3
  \end{bmatrix}$
  \end{problem}
   
  \begin{problem}\label{prob:notsimilar_e}
  $A =
  \begin{bmatrix}
  1 & 2 & -3 \\
  1 & -1 & 2 \\
  0 & 3 & -5
  \end{bmatrix}$, $B =
  \begin{bmatrix}
  -2 & 1 & 3 \\
   6 & -3 & -9 \\
   0 & 0 & 0
  \end{bmatrix}$
  \end{problem}
   
  \end{problem}
  
   
  
   
  \begin{problem}\label{prob:notsimilar_4x4}
  Show that $\begin{bmatrix}
  1 & 2 & -1 &  0 \\
  2 & 0 &  1 &  1 \\
  1 & 1 &  0 & -1 \\
  4 & 3 & 0 & 0
  \end{bmatrix}$ and $
  \begin{bmatrix}
    1 & -1 &  3 &  0 \\
   -1 &  0 &  1 &  1 \\
    0 & -1 &  4 &  1 \\
    5 & -1 & -1 & -4
  \end{bmatrix}$ are \textit{not} similar.
  \end{problem}
