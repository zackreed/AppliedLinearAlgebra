\section*{SVD, Eigenvalues and Eigenvectors}

\begin{example}

        \item The rank of $A$ is $\answer{2}$, so $A$ has $\answer{2}$ nonzero singular values.

        \item The null space of $A$ is $\answer{0}$-dimensional; $A\vec{x}=\vec{0}$.
    \end{enumerate}

\begin{example}
    Find the eigenvalues and eigenvectors of the matrix

    $A = \begin{bmatrix} 3 & 1 \\ 0 & 2 \end{bmatrix}$.

    \textbf{Answer:}
    The characteristic polynomial is $\det(A - \lambda I) = (\answer{3}-\lambda)(\answer{2}-\lambda) = 0$, so the eigenvalues are $\lambda_1 = \answer{3}$ and $\lambda_2 = \answer{2}$.

    For $\lambda_1 = 3$: An eigenvector is $\vec{v}_1 = \begin{bmatrix} \answer{1} \\ \answer{0} \end{bmatrix}$.

    For $\lambda_2 = 2$: An eigenvector is $\vec{v}_2 = \begin{bmatrix} \answer{1} \\ \answer{-1} \end{bmatrix}$.
\end{example}

\begin{example}

\end{example}

\section*{Multiplicity and Eigenspace}

\begin{example}
    Let $A = \begin{bmatrix} 2 & 1 & 0 \\ 0 & 2 & 0 \\ 0 & 0 & 3 \end{bmatrix}$.
    \begin{enumerate}
        \item The eigenvalues of $A$ (in increasing order) are $\lambda_1 = \answer{2}$ and $\lambda_2 = \answer{3}$.

        \item $\lambda_1$ has algebraic multiplicity $\answer{2}$ and geometric multiplicity $\answer{1}$. So the eigenspace for $\lambda_1$ is $\answer{1}$-dimensional.

        \item $\lambda_2$ has algebraic multiplicity $\answer{1}$ and geometric multiplicity $\answer{1}$.

        \item A basis for the eigenspace of $\lambda_1 $ is $\left\{\begin{bmatrix} \answer{1} \\ \answer{0} \\ \answer{0} \end{bmatrix}\right\}$.

        \item A basis for the eigenspace of $\lambda_2 $ is $\left\{\begin{bmatrix} \answer{0} \\ \answer{0} \\ \answer{1} \end{bmatrix}\right\}$.

        \item Is $A$ diagonalizable? \wordChoice{\choice{Yes, because all eigenvalues are distinct.}\choice[correct]{No, because an eigenvalue has algebraic multiplicity greater than its geometric multiplicity.}}.
    \end{enumerate}
\end{example}
